form.addEventListener('submit', async e=>{ e.preventDefault(); const name = document.getElementById('apkName').value; const apkFile = document.getElementById('apkFile').files[0]; const iconFile = document.getElementById('apkIcon').files[0]; if(!name || !apkFile || !iconFile) return; status.innerText = "Uploading..."; const progressBar = document.getElementById('progressBar'); // Upload APK with progress const apkRef = storage.ref('apks/' + apkFile.name); const apkUploadTask = apkRef.put(apkFile); apkUploadTask.on('state_changed', snapshot => { const progress = (snapshot.bytesTransferred / snapshot.totalBytes) * 100; progressBar.style.width = progress + '%'; status.innerText = 'Uploading APK: ' + Math.floor(progress) + '%'; }, error => { console.error(error); status.innerText = 'Upload failed!'; }, async () => { const apkURL = await apkUploadTask.snapshot.ref.getDownloadURL(); // Upload icon with progress const iconRef = storage.ref('images/' + iconFile.name); const iconUploadTask = iconRef.put(iconFile); iconUploadTask.on('state_changed', snapshot => { const progress = (snapshot.bytesTransferred / snapshot.totalBytes) * 100; progressBar.style.width = progress + '%'; status.innerText = 'Uploading Icon: ' + Math.floor(progress) + '%'; }, error => { console.error(error); status.innerText = 'Upload failed!'; }, async () => { const iconURL = await iconUploadTask.snapshot.ref.getDownloadURL(); await db.collection('apks').add({ name: name, apkURL: apkURL, iconURL: iconURL, timestamp: firebase.firestore.FieldValue.serverTimestamp() }); status.innerText = "Upload complete!"; progressBar.style.width = '100%'; form.reset(); } ); } ); });
